## Task 2: 在 GameEngine 中处理腐蚀

**Files:**
- Modify: `src/game/GameEngine.js`

**Step 1: 添加腐蚀效果处理方法**

```javascript
/**
 * 腐蚀手牌中的卡牌
 * @param {object} player - 玩家
 * @param {object} corruptCard - 腐蚀卡
 */
corruptHand(player, corruptCard) {
  if (!player.hand || player.hand.length === 0) {
    return;
  }

  const corruptEffect = corruptCard.effect.corrupt;
  if (!corruptEffect) return;

  // 查找可以被腐蚀的目标卡
  // 腐蚀条件：目标卡费用 <= 腐蚀卡费用
  const corruptCost = corruptCard.cost;
  const corruptTargets = player.hand.filter(card => {
    return card.cost <= corruptCost && !card.isCorrupted;
  });

  if (corruptTargets.length === 0) {
    Logger.info('没有可以被腐蚀的卡牌');
    return;
  }

  // 随机选择一个目标进行腐蚀
  const targetIndex = Math.floor(Math.random() * corruptTargets.length);
  const target = corruptTargets[targetIndex];

  // 应用腐蚀效果
  target.isCorrupted = true;
  target.corruptedEffect = corruptEffect;
  target.name = `${target.name} (已腐蚀)`;

  // 应用增强效果
  if (corruptEffect.attack) target.attack += corruptEffect.attack;
  if (corruptEffect.health) {
    target.health += corruptEffect.health;
    target.maxHealth = Math.max(target.maxHealth || target.health, target.health);
  }
  if (corruptEffect.taunt) target.taunt = true;
  if (corruptEffect.divine_shield) target.divine_shield = true;
  if (corruptEffect.windfury) target.windfury = true;
  if (corruptEffect.charge) target.charge = true;

  Logger.info(`${target.name} 被腐蚀，获得了强化效果`);

  // 如果被腐蚀的是随从卡，需要刷新其战场效果
  if (target.type === 'minion') {
    this.refreshMinionEffects(player, target);
  }
}

/**
 * 刷新随从效果
 */
refreshMinionEffects(player, minion) {
  // 确保随从在战场上的效果也更新
  const fieldMinion = player.field.find(m => m.id === minion.id);
  if (fieldMinion) {
    if (minion.corruptedEffect) {
      fieldMinion.isCorrupted = true;
      fieldMinion.corruptedEffect = minion.corruptedEffect;
      if (minion.corruptedEffect.attack) fieldMinion.attack += minion.corruptedEffect.attack;
      if (minion.corruptedEffect.health) {
        fieldMinion.health += minion.corruptedEffect.health;
        fieldMinion.maxHealth = Math.max(fieldMinion.maxHealth, fieldMinion.health);
      }
      if (minion.corruptedEffect.taunt) fieldMinion.taunt = true;
      if (minion.corruptedEffect.divine_shield) fieldMinion.divine_shield = true;
      if (minion.corruptedEffect.windfury) fieldMinion.windfury = true;
      if (minion.corruptedEffect.charge) fieldMinion.charge = true;
    }
  }
}
```
