## Task 2: 实现激励效果处理

**Files:**
- Modify: `src/game/CardEffect.js`

**Step 1: 添加 executeInspire 方法**

```javascript
executeInspire(effect, context) {
  // 激励效果在英雄技能使用后触发
  switch (effect.action) {
    case 'damage':
      const opponent = context.player === this.game.state.player
        ? this.game.state.ai
        : this.game.state.player;
      opponent.health -= effect.value;
      Logger.info(`激励造成 ${effect.value} 点伤害`);
      break;
    case 'heal':
      context.player.health = Math.min(
        context.player.health + effect.value,
        context.player.maxHealth
      );
      Logger.info(`激励恢复 ${effect.value} 点生命值`);
      break;
    case 'summon':
      if (effect.card_id) {
        const CardData = require('../data/CardData');
        const card = CardData.getCard(effect.card_id);
        if (card) {
          this.game.summonMinion(context.player, card);
        }
      } else if (effect.attack && effect.health) {
        // 直接召唤指定属性的随从
        const summonCard = {
          id: `inspire_${Date.now()}`,
          name: '激励召唤',
          type: 'minion',
          effect: { type: 'summon', attack: effect.attack, health: effect.health }
        };
        this.game.summonMinion(context.player, summonCard);
      }
      break;
    case 'armor':
      context.player.armor += effect.value;
      Logger.info(`激励获得 ${effect.value} 点护甲`);
      break;
    case 'draw_card':
      this.game.drawCard(context.player, effect.value || 1);
      break;
    case 'buff':
      // 激励buff所有友方随从
      context.player.field.forEach(m => {
        if (effect.attack) m.attack += effect.attack;
        if (effect.health) m.health += effect.health;
      });
      Logger.info('激励buff了友方随从');
      break;
    case 'destroy':
      // 激励摧毁一个随机敌方随从
      const targets = context.player === this.game.state.player
        ? this.game.state.ai.field
        : this.game.state.player.field;
      if (targets.length > 0) {
        const idx = Math.floor(Math.random() * targets.length);
        targets[idx].health = 0;
        Logger.info(`激励摧毁了 ${targets[idx].name}`);
      }
      break;
    default:
      Logger.warn(`未知的激励动作: ${effect.action}`);
  }
  return true;
}
```

添加 triggerInspire 方法:

```javascript
/**
 * 触发所有友方随从的激励效果
 */
triggerInspire(player) {
  const CardEffect = require('./CardEffect');
  const cardEffect = new CardEffect(this);

  player.field.forEach(minion => {
    if (minion.inspire) {
      cardEffect.execute(minion.inspire, { player, card: minion });
      Logger.info(`${minion.name} 的激励效果被触发`);
    }
  });
}
```
