## Task 2: 实现任务管理

**Files:**
- Create: `src/game/QuestManager.js`

**Step 1: 创建任务管理器**

```javascript
// src/game/QuestManager.js
class QuestManager {
  constructor(game) {
    this.game = game;
  }

  /**
   * 初始化任务
   * @param {object} quest - 任务卡牌
   * @param {object} player - 玩家
   */
  initQuest(quest, player) {
    if (!player.activeQuests) {
      player.activeQuests = [];
    }

    player.activeQuests.push({
      card: quest,
      progress: 0,
      target: quest.effect.questProgress || 1,
      condition: quest.effect.questCondition || 'play_card',
      reward: quest.effect.reward
    });

    Logger.info(`${player.name} 完成了任务: ${quest.name}`);
  }

  /**
   * 更新任务进度
   * @param {object} player - 玩家
   * @param {string} condition - 触发条件
   * @param {object} data - 额外数据
   */
  updateProgress(player, condition, data) {
    if (!player.activeQuests || player.activeQuests.length === 0) {
      return;
    }

    const completed = [];

    player.activeQuests.forEach((quest, index) => {
      if (quest.condition === condition) {
        quest.progress++;

        // 检查是否完成
        if (quest.progress >= quest.target) {
          completed.push(quest);
        }

        Logger.info(`任务进度: ${quest.progress}/${quest.target}`);
      }
    });

    // 处理完成的任务
    completed.forEach(quest => {
      this.completeQuest(player, quest);
      // 移除已完成任务
      player.activeQuests = player.activeQuests.filter(q => q !== quest);
    });
  }

  /**
   * 完成任务，发放奖励
   * @param {object} player - 玩家
   * @param {object} quest - 任务
   */
  completeQuest(player, quest) {
    Logger.info(`任务完成: ${quest.card.name}, 发放奖励`);

    const CardEffect = require('./CardEffect');
    const cardEffect = new CardEffect(this.game);

    const reward = quest.reward;

    if (reward.type === 'summon') {
      // 召唤奖励随从
      if (reward.card_id) {
        const CardData = require('../data/CardData');
        const card = CardData.getCard(reward.card_id);
        if (card) {
          this.game.summonMinion(player, card);
        }
      } else if (reward.attack && reward.health) {
        const rewardCard = {
          id: `reward_${Date.now()}`,
          name: reward.name || '任务奖励',
          type: 'minion',
          effect: { type: 'summon', attack: reward.attack, health: reward.health }
        };
        this.game.summonMinion(player, rewardCard);
      }
    } else if (reward.type === 'spell') {
      // 将奖励法术置入手中
      if (reward.card_id) {
        const CardData = require('../data/CardData');
        const card = CardData.getCard(reward.card_id);
        if (card) {
          this.game.addCardToHand(player, card);
        }
      }
    } else if (reward.type === 'damage') {
      // 直接造成伤害
      const opponent = player === this.game.state.player
        ? this.game.state.ai
        : this.game.state.player;
      opponent.health -= reward.value;
    } else if (reward.type === 'buff') {
      // Buff 所有友方随从
      player.field.forEach(m => {
        if (reward.attack) m.attack += reward.attack;
        if (reward.health) m.health += reward.health;
      });
    }
  }
}

module.exports = QuestManager;
```

```javascript
// 检查是否有任务卡
if (card.effect?.type === 'quest') {
  this.questManager.initQuest(card, player);
  // 任务卡不进入战场，直接消耗
  return this.getGameState();
}

// 触发任务进度
if (card.type === 'minion') {
  this.questManager.updateProgress(player, 'play_minion', { card });
} else if (card.type === 'spell') {
  this.questManager.updateProgress(player, 'play_spell', { card });
}
```
