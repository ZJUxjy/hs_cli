## Task 1: 创建 RL API Server 基础框架

**Files:**

**Step 1: 创建 RLServer 类框架**

```javascript
const express = require('express');

class RLServer {
  constructor(gameEngine, port = 3000) {
    this.app = express();
    this.game = gameEngine;
    this.port = port;
    this.currentState = null;
    this.lastReward = 0;

    this.app.use(express.json());

    // 注册路由
    this.registerRoutes();
  }

  registerRoutes() {
    this.app.post('/reset', this.reset.bind(this));
    this.app.post('/step', this.step.bind(this));
    this.app.get('/action_space', this.getActionSpace.bind(this));
    this.app.get('/observation_space', this.getObservationSpace.bind(this));
    this.app.get('/game_state', this.getGameState.bind(this));
  }

  start() {
    return new Promise(resolve => {
      this.app.listen(this.port, () => {
        console.log(`RL Server running on port ${this.port}`);
        resolve();
      });
    });
  }
}

module.exports = RLServer;
```

```javascript
async reset(req, res) {
  const { player_class = 'mage', opponent_class = 'warrior' } = req.body;

  // 初始化新游戏
  this.game.startNewGame(player_class, 'normal');

  this.currentState = this.game.getGameState();
  this.lastReward = 0;

  const response = {
    observation: this.getObservation(),
    action_mask: this.getActionMask(),
    reward: 0,
    done: false,
    info: {}
  };

  res.json(response);
}

getObservation() {
  const state = this.currentState;

  // 简化状态表示
  return {
    player: {
      health: state.player.health,
      armor: state.player.armor,
      mana: state.player.mana,
      maxMana: state.player.maxMana,
      hand_count: state.player.hand.length,
      field_count: state.player.field.length,
      deck_count: state.player.deck.length
    },
    opponent: {
      health: state.ai.health,
      armor: state.ai.armor,
      hand_count: state.ai.hand.length,
      field_count: state.ai.field.length,
      deck_count: state.ai.deck.length
    },
    turn: state.turn,
    game_phase: state.phase
  };
}
```

```javascript
async step(req, res) {
  const { action } = req.body;

  if (!action) {
    return res.status(400).json({ error: 'action required' });
  }

  const state = this.currentState;

  // 检查游戏是否结束
  if (state.phase === 'ended') {
    return res.json({
      observation: this.getObservation(),
      action_mask: [],
      reward: 0,
      done: true,
      info: { winner: state.winner }
    });
  }

  // 计算执行动作前的奖励
  const damageDealt = this.calculateReward(state, action);

  // 执行动作
  this.executeAction(action);

  // 获取新状态
  this.currentState = this.game.getGameState();
  const newState = this.currentState;

  // 计算奖励
  const reward = this.calculateReward(state, action);

  // 检查游戏结束
  const done = newState.phase === 'ended';

  res.json({
    observation: this.getObservation(),
    action_mask: this.getActionMask(),
    reward: reward,
    done: done,
    info: {}
  });
}
```

**Step 2: 添加动作执行方法**

```javascript
executeAction(action) {
  const state = this.currentState;
  const player = state.player;

  switch (action.type) {
    case 'play_card':
      const card = player.hand[action.card_index];
      if (card && player.mana >= card.cost) {
        this.game.removeCardFromHand(player, card);
        player.mana -= card.cost;

        if (card.type === 'minion') {
          this.game.summonMinion(player, card);
        } else {
          // 法术效果需要目标
          const target = action.target === 'hero'
            ? state.ai
            : state.ai.field[action.target];
          const CardEffect = require('../game/CardEffect');
          const ce = new CardEffect(this.game);
          ce.execute(card, { player, target, card });
        }
      }
      break;

    case 'attack':
      const attacker = player.field[action.attacker];
      if (attacker && attacker.canAttack && !attacker.hasAttacked) {
        const target = action.target === 'hero'
          ? state.ai
          : state.ai.field[action.target];

        if (target.health !== undefined) {
          target.health -= attacker.attack;
          attacker.health -= target.attack;
        } else {
          state.ai.health -= attacker.attack;
        }
        attacker.hasAttacked = true;
      }
      break;

    case 'end_turn':
      this.game.switchTurn();
      break;

    case 'hero_power':
      // 简化：直接使用英雄技能
      player.usedHeroPower = true;
      player.mana -= 2;
      break;
  }

  // 清理死亡随从
  this.game.removeDeadMinions();
}
```

**Step 3: 添加奖励计算**

```javascript
calculateReward(oldState, action) {
  const newState = this.game.getGameState();

  let reward = 0;

  // 造成伤害奖励
  const damage = oldState.ai.health - newState.ai.health;
  if (damage > 0) {
    reward += damage / 10;
  }

  // 胜负奖励
  if (newState.phase === 'ended') {
    if (newState.winner === 'player') {
      reward += 100;
    } else {
      reward -= 100;
    }
  }

  return reward;
}
```

```javascript
getActionSpace(req, res) {
  // 动作空间：
  // 0-9: 出手牌 (0-9号手牌)
  // 10-29: 攻击 (attacker*7 + target, target=0=英雄, 1-7=随从)
  // 30: 结束回合
  // 31: 使用英雄技能

  const actions = [];

  // 出手牌
  for (let i = 0; i < 10; i++) {
    actions.push({ id: i, type: 'play_card', card_index: i });
  }

  // 攻击
  for (let attacker = 0; attacker < 7; attacker++) {
    // 攻击英雄
    actions.push({ id: 10 + attacker * 8, type: 'attack', attacker, target: 'hero' });
    // 攻击随从
    for (let target = 0; target < 7; target++) {
      actions.push({ id: 10 + attacker * 8 + target + 1, type: 'attack', attacker, target });
    }
  }

  // 结束回合
  actions.push({ id: 66, type: 'end_turn' });

  // 英雄技能
  actions.push({ id: 67, type: 'hero_power' });

  res.json({
    n_actions: actions.length,
    actions: actions
  });
}
```

```javascript
getActionMask() {
  const state = this.currentState;
  const player = state.player;
  const ai = state.ai;

  const mask = new Array(68).fill(0);

  // 出手牌
  player.hand.forEach((card, i) => {
    if (i < 10 && player.mana >= card.cost) {
      mask[i] = 1;
    }
  });

  // 攻击
  player.field.forEach((attacker, attackerIdx) => {
    if (attacker.canAttack && !attacker.hasAttacked && !attacker.sleeping && !attacker.frozen) {
      // 可以攻击英雄
      mask[10 + attackerIdx * 8] = 1;

      // 可以攻击随从
      const hasTaunt = ai.field.some(m => m.taunt);
      ai.field.forEach((target, targetIdx) => {
        if (!hasTaunt || target.taunt) {
          mask[10 + attackerIdx * 8 + targetIdx + 2] = 1;
        }
      });
    }
  });

  // 结束回合总是可用
  mask[66] = 1;

  // 英雄技能
  if (!player.usedHeroPower && player.mana >= 2) {
    mask[67] = 1;
  }

  return mask;
}
```
