const router = express.Router();
const DeckBuilder = require('../../data/DeckBuilder');

router.get('/', (req, res) => {
  res.json(DeckBuilder.list());
});

router.post('/', (req, res) => {
  const { name, hero, mode } = req.body;
  const deck = DeckBuilder.create(name, hero, mode || 'standard');
  res.json(deck);
});

router.get('/:id', (req, res) => {
  const deck = DeckBuilder.getDeck(req.params.id);
  if (deck) {
    res.json(deck);
  } else {
    res.status(404).json({ error: 'Deck not found' });
  }
});

router.put('/:id', (req, res) => {
  const deck = DeckBuilder.update(req.params.id, req.body);
  res.json(deck);
});

router.delete('/:id', (req, res) => {
  DeckBuilder.delete(req.params.id);
  res.json({ success: true });
});

router.post('/:id/cards', (req, res) => {
  const { cardId } = req.body;
  const deck = DeckBuilder.addCard(req.params.id, cardId);
  res.json(deck);
});

module.exports = router;

```javascript
// public/js/api.js
const API = {
  async getCards() {
    const res = await fetch('/api/cards');
    return res.json();
  },

  async getCardsByClass(cls) {
    const res = await fetch(`/api/cards/class/${cls}`);
    return res.json();
  },

  async getDecks() {
    const res = await fetch('/api/decks');
    return res.json();
  },

  async createDeck(name, hero, mode) {
    const res = await fetch('/api/decks', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ name, hero, mode })
    });
    return res.json();
  },

  async startGame(playerClass, opponentClass) {
    const res = await fetch('/api/game/start', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ playerClass, opponentClass })
    });
    return res.json();
  },

  async getGameState() {
    const res = await fetch('/api/game/state');
    return res.json();
  },

  async playCard(cardIndex) {
    const res = await fetch('/api/game/play', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ cardIndex })
    });
    return res.json();
  },

  async endTurn() {
    const res = await fetch('/api/game/endTurn', {
      method: 'POST'
    });
    return res.json();
  }
};
```

// public/js/app.js
class App {
  constructor() {
    this.currentScreen = 'menu';

    Object.values(this.screens).forEach(s => s.classList.add('hidden'));

  }
}

// public/js/deck.js
class DeckBuilderUI {
  constructor() {
    this.currentDeck = { name: '', hero: 'mage', cards: [] };
    this.allCards = [];
  }

  async init() {
    this.allCards = await API.getCards();
    this.renderCardList(this.allCards);
    this.bindEvents();
  }

  renderCardList(cards) {
    const list = document.getElementById('card-list');
    list.innerHTML = cards.slice(0, 100).map(card => `
      <div class="card-item" data-id="${card.id}" draggable="true">
        <div class="card-name">${card.name}</div>
        <div class="card-cost">${card.cost}</div>
      </div>
    `).join('');
  }

  bindEvents() {
    // 搜索
    document.getElementById('card-search').addEventListener('input', (e) => {
      const q = e.target.value.toLowerCase();
      const filtered = this.allCards.filter(c =>
        c.name && c.name.toLowerCase().includes(q)
      );
      this.renderCardList(filtered);
    });

    // 拖拽
    document.getElementById('card-list').addEventListener('dragstart', (e) => {
      e.dataTransfer.setData('cardId', e.target.dataset.id);
    });

    // 保存卡组
    document.getElementById('btn-save-deck').addEventListener('click', async () => {
      const name = document.getElementById('deck-name').value;
      const hero = document.getElementById('deck-hero').value;
      await API.createDeck(name, hero, 'standard');
      alert('卡组保存成功!');
    });
  }
}

document.addEventListener('DOMContentLoaded', () => {
  window.deckBuilder = new DeckBuilderUI();
  window.deckBuilder.init();
});

// public/js/game.js
class GameUI {
  constructor() {
    this.gameState = null;
  }

  async startGame(playerClass) {
    this.gameState = await API.startGame(playerClass, 'warrior');
    this.render();
  }

  render() {
    const state = this.gameState;
    if (!state) return;

    // 渲染玩家信息
    document.getElementById('player-hero').innerHTML = `
      <div>生命: ${state.player.health}</div>
      <div>法力: ${state.player.mana}/${state.player.maxMana}</div>
    `;

    // 渲染手牌
    const handEl = document.getElementById('player-hand');
    handEl.innerHTML = state.player.hand.map((card, i) => `
      <div class="card" data-index="${i}">
        <div>${card.name}</div>
        <div>${card.cost}</div>
      </div>
    `).join('');

    // 渲染战场
    this.renderField('enemy-field', state.ai.field);
    this.renderField('player-field', state.player.field);
  }

  renderField(elementId, minions) {
    const el = document.getElementById(elementId);
    el.innerHTML = minions.map((m, i) => `
      <div class="minion" data-index="${i}">
        <div>${m.attack}/${m.health}</div>
        <div>${m.name}</div>
      </div>
    `).join('');
  }

  bindEvents() {
    document.getElementById('btn-end-turn').addEventListener('click', async () => {
      this.gameState = await API.endTurn();
      this.render();
    });

    document.getElementById('player-hand').addEventListener('click', async (e) => {
      const cardEl = e.target.closest('.card');
      if (cardEl) {
        const index = parseInt(cardEl.dataset.index);
        this.gameState = await API.playCard(index);
        this.render();
      }
    });
  }
}
