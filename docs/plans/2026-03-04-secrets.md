## Task 2: 实现奥秘效果执行

**Files:**
- Modify: `src/game/GameEngine.js`

**Step 1: 扩展 triggerSecret 方法**

```javascript
triggerSecret(secret, data) {
  const effect = secret.effect || secret.originalCard?.effect;

  if (!effect || !effect.type) {
    Logger.warn('奥秘没有效果:', secret.originalCard?.name);
    return;
  }

  switch (effect.type) {
    case 'damage':
      // 对敌方造成伤害
      const opponent = this.getOpponent();
      opponent.health -= effect.value;
      Logger.info(`奥秘造成 ${effect.value} 点伤害`);
      break;
    case 'aoe_damage':
      // 对所有敌人造成伤害
      const opp = this.getOpponent();
      opp.field.forEach(m => m.health -= effect.value);
      Logger.info(`奥秘对所有敌人造成 ${effect.value} 点伤害`);
      break;
    case 'freeze':
      // 冻结敌人
      const o = this.getOpponent();
      o.frozen = true;
      Logger.info('奥秘冻结敌方英雄');
      break;
    case 'secret':
      // 直接执行奥秘内的效果
      const CardEffect = require('./CardEffect');
      const cardEffect = new CardEffect(this);
      cardEffect.execute(secret.originalCard, { player: this.state.player });
      break;
    case 'summon':
      // 召唤随从
      if (effect.id) {
        const CardData = require('../data/CardData');
        const card = CardData.getCard(effect.id);
        if (card) {
          this.summonMinion(this.state.player, card);
        }
      }
      break;
    case 'buff':
      // 增益友方随从
      this.state.player.field.forEach(m => {
        if (effect.attack) m.attack += effect.attack;
        if (effect.health) m.health += effect.health;
      });
      Logger.info('奥秘 buff 了友方随从');
      break;
    case 'return':
      // 返回手牌
      if (data && data.target) {
        this.bounceMinion(data.target);
      }
      break;
    case 'destroy':
      // 摧毁随从
      if (data && data.target) {
        data.target.health = 0;
      }
      break;
    case 'gain_armor':
      // 获得护甲
      this.state.player.armor += (effect.value || 0);
      Logger.info(`奥秘给予 ${effect.value} 点护甲`);
      break;
  }
}
```
