**Step 2: 修改 removeDeadMinions 处理复生**

在移除死亡随从前添加复生逻辑:

```javascript
removeDeadMinions() {
  if (!this.state) return;

  const player = this.state.player;
  const ai = this.state.ai;

  // 分离死亡和存活的随从
  const deadPlayerMinions = player.field.filter(m => m.health <= 0);
  const deadAiMinions = ai.field.filter(m => m.health <= 0);

  // 处理复生
  const rebornPlayerMinions = [];
  const rebornAiMinions = [];

  deadPlayerMinions.forEach(minion => {
    if (minion.reborn && !minion.hasReborn) {
      // 复生：1血复活
      minion.health = 1;
      minion.maxHealth = Math.max(minion.maxHealth, 1);
      minion.hasReborn = true; // 标记已使用复生
      minion.reborn = false;   // 移除复生标记
      rebornPlayerMinions.push(minion);
      Logger.info(`${minion.name} 触发复生效果，1血复活！`);
    }
  });

  deadAiMinions.forEach(minion => {
    if (minion.reborn && !minion.hasReborn) {
      minion.health = 1;
      minion.maxHealth = Math.max(minion.maxHealth, 1);
      minion.hasReborn = true;
      minion.reborn = false;
      rebornAiMinions.push(minion);
      Logger.info(`${minion.name} 触发复生效果，1血复活！`);
    }
  });

  // 移除真正死亡的随从
  player.field = player.field.filter(m => m.health > 0);
  ai.field = ai.field.filter(m => m.health > 0);

  // 将复生随从重新放回战场
  rebornPlayerMinions.forEach(m => player.field.push(m));
  rebornAiMinions.forEach(m => ai.field.push(m));

  // 触发亡语（复生不触发亡语）
  deadPlayerMinions.forEach(minion => {
    if (minion.deathrattle && !minion.hasReborn) {
      // 亡语处理...
    }
  });

  // ... 原有代码
}
```
