在 playCard 方法中，法术或随从打出后:

```javascript
// 检查回响效果
if (card.echo) {
  if (!player.echoCards) {
    player.echoCards = [];
  }
  // 复制卡牌数据（带回响标记）
  const echoCard = {
    ...card,
    id: `${card.id}_echo_${Date.now()}`,

    echo: { copies: card.echo.copies || 1 }
  };

```

**Step 2: 添加回合结束回响处理**

在 endTurn 方法中:

```javascript
// 处理回响效果
this.processEchoCards(player);

/**
 * 处理回响卡牌
 */
processEchoCards(player) {
  if (!player.echoCards || player.echoCards.length === 0) {
    return;
  }

  const copiedCards = [];
  player.echoCards.forEach(card => {
    const copies = card.echo?.copies || 1;
    for (let i = 0; i < copies; i++) {
      const copyCard = { ...card };
      copyCard.id = `${card.id}_copy_${Date.now()}_${i}`;
      copyCard.name = card.name.replace(' (回响)', '');
      copyCard.echo = false; // 复制的卡不再有回响
      copiedCards.push(copyCard);
    }
  });

  // 将复制的手牌加入手牌
  copiedCards.forEach(card => {
    if (player.hand.length < 10) {
      player.hand.push(card);
      Logger.info(`回响复制了 ${card.name} 到手牌`);
    } else {
      Logger.info(`手牌已满，无法复制 ${card.name}`);
    }
  });

  // 清空回响列表
  player.echoCards = [];
}
```
