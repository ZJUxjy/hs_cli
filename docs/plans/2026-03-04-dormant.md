```javascript
// 存储战吼、亡语、激励和休眠
minion.battlecry = card.effect?.battlecry || null;
minion.deathrattle = card.effect?.deathrattle || null;

}
```

**Step 2: 添加唤醒逻辑**

在回合结束或回合开始时检查休眠:

```javascript
/**
 * 检查并唤醒休眠的随从
 */
checkDormantWakeup(player) {
  if (!player.field) return;

  player.field.forEach(minion => {
    if (minion.dormant && !minion.dormant.awakened) {
      minion.dormant.turns--;

      if (minion.dormant.turns <= 0) {
        // 唤醒随从
        minion.dormant.awakened = true;
        minion.sleeping = false;
        minion.canAttack = true;
        Logger.info(`${minion.name} 醒来了！`);

        // 执行唤醒效果
        if (minion.dormant.wakeEffect) {
          this.executeWakeEffect(minion, minion.dormant.wakeEffect);
        }
      } else {
        Logger.info(`${minion.name} 还需要 ${minion.dormant.turns} 回合唤醒`);
      }
    }
  });
}

/**
 * 执行唤醒效果
 */
executeWakeEffect(minion, wakeEffect) {
  const CardEffect = require('./CardEffect');
  const cardEffect = new CardEffect(this);

  const player = this.state.player.field.includes(minion)
    ? this.state.player
    : this.state.ai;

  switch (wakeEffect.type) {
    case 'damage':
      const opponent = player === this.state.player ? this.state.ai : this.state.player;
      opponent.health -= wakeEffect.value;
      Logger.info(`唤醒效果造成 ${wakeEffect.value} 点伤害`);
      break;
    case 'aoe_damage':
      const opp = player === this.state.player ? this.state.ai : this.state.player;
      opp.field.forEach(m => m.health -= wakeEffect.value);
      Logger.info(`唤醒效果对所有敌人造成 ${wakeEffect.value} 点伤害`);
      break;
    case 'buff':
      player.field.forEach(m => {
        if (wakeEffect.attack) m.attack += wakeEffect.attack;
        if (wakeEffect.health) m.health += wakeEffect.health;
      });
      Logger.info('唤醒效果buff了友方随从');
      break;
    case 'summon':
      if (wakeEffect.card_id) {
        const CardData = require('../data/CardData');
        const card = CardData.getCard(wakeEffect.card_id);
        if (card) {
          this.summonMinion(player, card);
        }
      }
      break;
    default:
      Logger.warn(`未知的唤醒效果类型: ${wakeEffect.type}`);
  }
}
```
